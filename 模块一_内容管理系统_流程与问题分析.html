📋 模块概览

内容管理系统是 RFYR 平台的核心模块，负责管理所有数字化内容的生命周期，包括创建、编辑、存储、访问控制和展示。

核心职责

  • 文章/笔记的 CRUD 操作（创建、读取、更新、删除）
  • 分类体系的树形结构管理
  • 多种内容格式支持（富文本、PDF、HTML 嵌入）
  • 访问权限控制（free/monthly/yearly）
  • 阅读量统计和追踪
  • 内容展示和导航

步骤 5: 生成 Short ID
// lib/short-id.ts
export function generateShortId(length = 8): string {
  const bytes = randomBytes(length)
  let result = ""
  for (let i = 0; i < length; i++) {
    result += CHARS[bytes[i] % CHARS.length]
  }
  return result
}
步骤 6: 提交到 API

表单提交后，通过 createArticle() 函数调用 Supabase 插入数据。

// 3. 游客检查

  if (article.access_level === "free") return article.content
  return { error: "请先登录", code: "REQUIRE_LOGIN" }, 401
}

// 4. 会员类型检查
const isYearly = await isYearlyMember(userId)
const isMonthly = await isMonthlyMember(userId)

// 5. 文章权限级别检查
if (article.access_level === "yearly" && !isYearly) {
  return { error: "年卡专属", code: "YEARLY_REQUIRED" }, 403
}
if (article.access_level === "monthly" && !isMonthly && !isYearly) {
  return { error: "需要会员", code: "MEMBERSHIP_REQUIRED" }, 403
}

// 6. 配额检查（原子 upsert）
if (isMonthly) {
  const result = await recordUserReadAtomic(userId, article.id, dailyLimit, today)
  if (result === "exceeded") return { error: "每日限额", code: "DAILY_LIMIT_EXCEEDED" }, 403
} else if (!isYearly) {
  const result = await recordNonMemberReadAtomic(userId, article.id, lifetimeLimit, today)
  if (result === "exceeded") return { error: "配额用完", code: "LIMIT_EXCEEDED" }, 403
}

// 7. 返回内容
return article.content

// handlePaste 中的图片处理逻辑

存储路径安全验证

// RichEditor.tsx
function isSafeStorageObjectPathForApi(path: string): boolean {
  if (!path || path.length > 1024) return false  // 长度限制
  if (path.includes("..")) return false         // 防止路径穿越
  if (path.startsWith("/")) return false        // 防止绝对路径
  return /^[a-zA-Z0-9._\-/]+$/.test(path)     // 白名单字符
}

  const companionUrls = await Promise.all(
    companions.map(f => uploadToTemp(f, tempDir))

  }

// lib/category-utils.ts
export interface CategoryNode {
  id: string
  name: string
  parentId?: string | null
  href?: string | null
}

export function buildCategoryMaps(rows: CategoryNode[]) {
  const categoryMap: Record<string, CategoryNode> = {}
  const nameToIdMap: Record<string, string> = {}
  const childrenMap: Record<string, string[]> = {}
  
  for (const row of rows) {
    categoryMap[row.id] = row
    if (row.name) nameToIdMap[row.name] = row.id
    
    const parentId = row.parentId || ''
    if (parentId) {
      if (!childrenMap[parentId]) childrenMap[parentId] = []
      childrenMap[parentId].push(row.id)
    }
  }
  
  return { categoryMap, nameToIdMap, childrenMap }
}

export function isInCategoryTree(
  articleCategory: string,
  rootCategoryName: string,
  categoryMap: Record<string, CategoryNode>,
  nameToIdMap: Record<string, string>
): boolean {
  if (articleCategory === rootCategoryName) return true
  
  const findParents = (name: string): string[] => {
    const id = nameToIdMap[name]
    if (!id) return []
    const parents: string[] = []
    let current = categoryMap[id]
    while (current?.parentId) {
      parents.push(current.parentId)
      current = categoryMap[current.parentId]
    }
    return parents
  }
  
  const parents = findParents(articleCategory)
  return parents.some(p => categoryMap[p]?.name === rootCategoryName)
}

export function filterArticlesByCategory(
  articles: Article[],
  rootCategoryName: string,
  categoryMap: Record<string, CategoryNode>,
  nameToIdMap: Record<string, string>
): Article[] {
  return articles.filter(a => isInCategoryTree(a.category, rootCategoryName, categoryMap, nameToIdMap))
}

// lib/articles.ts 使用
import { buildCategoryMaps, filterArticlesByCategory } from './category-utils'

export async function getArticlesByCategory(category: string): Promise<Article[]> {
  const categoriesData = await supabase.from('categories').select('*')
  const allArticles = await supabase.from('articles').select('*')
  
  const { categoryMap, nameToIdMap } = buildCategoryMaps(categoriesData)
  return filterArticlesByCategory(allArticles, category, categoryMap, nameToIdMap)
}

function getStorageKey(type: 'pdf' | 'html', articleId: string): string {
  const hash = simpleHash(articleId)
  return `${type}_original_name_${hash}`
}

function simpleHash(str: string): string {
  // 简单的确定性 hash，不泄露原始值
  let h = 0
  for (let i = 0; i < str.length; i++) {
    h = ((h << 5) - h) ^ str.charCodeAt(i)
  }
  return Math.abs(h).toString(36)
}

✓ 解决方案 7: 添加图片上传重试

async function uploadWithRetry(
  uploadFn: () => Promise<string>,
  maxRetries: number = 3,
  delayMs: number = 1000
): Promise<string> {
  let lastError: Error | null = null
  
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await uploadFn()
    } catch (e) {
      lastError = e as Error
      console.warn(`上传失败，${maxRetries - i - 1} 次重试中...`, e)
      
      if (i < maxRetries - 1) {
        // 指数退避
        await new Promise(r => setTimeout(r, delayMs * Math.pow(2, i)))
      }
    }
  }
  
  throw lastError || new Error('上传失败')
}

  })
