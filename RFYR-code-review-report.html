app/api/articles/[id]/route.ts:540–580
▼
问题描述

非会员用户读取文章时，代码先执行 SELECT 查询当前已读篇数（第540行），检查是否超限（第552行），然后执行 UPDATE 写入新的已读记录（第572行）。在 SELECT 和 UPDATE 之间存在竞态窗口——用户同时打开多个标签页时，每个请求都能通过检查，然后全部执行写入，导致 notes_read_count 超过终身限制 lifetimeLimit。

✅ 解决方案
方案：将非会员路径改为原子 upsert 逻辑

参考月卡路径的 recordUserReadAtomic 函数，在一次数据库操作中完成"检查超限 → 写入记录"，使用 Supabase RPC 或带条件更新的 upsert 实现原子性。

// 在 route.ts 顶部添加原子记录函数（非会员版本）
async function recordNonMemberReadAtomic(
  userId: string,
  articleId: string,
  lifetimeLimit: number,
  today: string
): Promise<"exceeded" | "already_read" | { readCount: number }> {
  const supabase = createClient(supabaseUrl, supabaseServiceKey)

  // 1. 读取当前状态
  const { data: profile } = await supabase
    .from("user_profiles")
    .select("notes_read_ids, notes_read_count")
    .eq("id", userId)
    .single()

  const existingIds: string[] = profile?.notes_read_ids ?? []

  // 2. 已读：不计数，直接放行
  if (existingIds.includes(articleId)) {
    return "already_read"
  }

  // 3. 检查是否超限（使用当前数据库值，不是传入值）
  const currentCount = existingIds.length
  if (currentCount >= lifetimeLimit) {
    return "exceeded"
  }

  // 4. 原子写入：只有当前 count < limit 时才写入
  // 使用条件更新确保原子性
  const newIds = [...existingIds, articleId]
  const { data: updateResult, error: updateError } = await supabase
    .from("user_profiles")
    .update({
      notes_read_count: newIds.length,
      notes_read_ids: newIds,
      last_read_date: today,
    })
    .eq("id", userId)
    .eq("notes_read_count", currentCount) // ← 关键：条件更新
    .select("notes_read_count")
    .single()

  // 5. 如果条件不满足（并发更新导致 count 已变），重新检查
  if (updateError || !updateResult) {
    // 条件更新失败，说明有并发，重新获取并判断
    const { data: retryProfile } = await supabase
      .from("user_profiles")
      .select("notes_read_ids, notes_read_count")
      .eq("id", userId)
      .single()

    if (retryProfile?.notes_read_ids?.includes(articleId)) {
      return "already_read"
    }
    return Number(retryProfile?.notes_read_count ?? 0) >= lifetimeLimit
      ? "exceeded"
      : { readCount: Number(retryProfile?.notes_read_count ?? 0) + 1 }
  }

  return { readCount: newIds.length }
}

#3 · 严重 · 安全
管理员认证存在弱 fallback，任何非空字符串可冒充管理员
lib/server-admin-auth.ts:27–30
▼
问题描述

当 admin-session-local cookie 解析失败或为空时，代码 fallback 到 admin-session cookie，并接受任何非空字符串作为管理员 ID（第28–29行）。如果攻击者能设置 admin-session cookie（值为任意用户 UUID），即可绕过管理员权限验证。

虽然 admin-session 设置了 httpOnly: true 防止 XSS 直接读取，但配合其他漏洞（如中间人攻击、代理篡改响应），仍存在风险。

✅ 解决方案
方案：移除不安全的 fallback，改为验证用户是否为管理员
// lib/server-admin-auth.ts 完全重写
import { NextRequest, NextResponse } from "next/server"
import { createClient } from "@supabase/supabase-js"

const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL!
const supabaseServiceKey = process.env.SUPABASE_SERVICE_ROLE_KEY!

const ADMIN_EMAILS = (process.env.ADMIN_EMAILS ?? "")
  .split(",")
  .map(e => e.trim().toLowerCase())

export async function requireAdmin(request: NextRequest): Promise<NextResponse | null> {
  const adminSessionLocal = request.cookies.get("admin-session-local")
  const adminSession = request.cookies.get("admin-session")

  let userId: string | null = null

  // 优先从 admin-session-local 解析
  if (adminSessionLocal?.value) {
    try {
      const session = JSON.parse(
        decodeURIComponent(adminSessionLocal.value)
      )
      if (session?.userId && typeof session.userId === "string") {
        userId = session.userId
      }
    } catch {
      // 解析失败，尝试 fallback
    }
  }

  // Fallback：必须有 HMAC 签名验证，不能仅依赖 cookie 存在
  // 移除对 admin-session 的无条件信任 fallback

  if (!userId) {
    return NextResponse.json(
      { error: "请先登录管理员账号" },
      { status: 401 }
    )
  }

  // 验证该用户是否为管理员（在数据库中查询 admin 权限）
  const supabase = createClient(supabaseUrl, supabaseServiceKey)
  const { data: user } = await supabase
    .from("users")
    .select("email, is_admin")
    .eq("id", userId)
    .maybeSingle()

  const isAdmin = user?.is_admin === true ||
    (ADMIN_EMAILS.length > 0 && ADMIN_EMAILS.includes(user?.email?.toLowerCase() ?? ""))

  if (!isAdmin) {
    return NextResponse.json(
      { error: "您没有后台管理权限" },
      { status: 403 }
    )
  }

  return null // 通过认证
}
💡 补充：在 users 表中添加 is_admin 布尔列，或在 Supabase 中创建 RLS 策略限制管理员操作。

#6 · 高 · 安全
recordVisit 函数存在 select-then-update 竞态
lib/reading-limit.ts:101–159
▼
问题描述

recordVisit 函数先 SELECT 读取当前已读列表，检查是否已读，然后 UPDATE 写入。整个过程不是原子操作，与问题#1类似，但影响月卡用户通过 POST /api/reading-limit 直接调用的场景。

✅ 解决方案
方案：参考 article API 中的原子 upsert 逻辑重构

使用条件更新（.eq("notes_read_count", currentCount)）确保并发安全，或使用 Supabase RPC 在数据库层面加锁。

// lib/reading-limit.ts — recordVisit 重构
export async function recordVisit(
  userId: string,
  articleId: string
): Promise<ReadingLimitData> {
  if (!supabaseAdmin) {
    return { readCount: 0, readIds: [], dailyReadCount: 0, lastReadDate: null }
  }

  const today = toLocalDateString()
  const { data: profile, error: fetchError } = await supabaseAdmin
    .from("user_profiles")
    .select("notes_read_ids, daily_read_count, last_read_date")
    .eq("id", userId)
    .single()

  if (fetchError && fetchError.code !== "PGRST116") {
    return { readCount: 0, readIds: [], dailyReadCount: 0, lastReadDate: null }
  }

  const existingIds: string[] = profile?.notes_read_ids ?? []

  // 已在阅读列表中——直接返回（幂等）
  if (existingIds.includes(articleId)) {
    const lastReadDate = profile?.last_read_date
      ? String(profile.last_read_date).split("T")[0]
      : null
    const shouldReset = lastReadDate !== today
    return {
      readCount: existingIds.length,
      readIds: existingIds,
      dailyReadCount: shouldReset ? 0 : Number(profile?.daily_read_count ?? 0),
      lastReadDate: profile?.last_read_date ?? null,
    }
  }

  const lastReadDate = profile?.last_read_date
    ? String(profile.last_read_date).split("T")[0]
    : null
  const shouldResetDaily = lastReadDate !== today
  const currentDailyCount = shouldResetDaily
    ? 0
    : Number(profile?.daily_read_count ?? 0)
  const newDailyCount = shouldResetDaily ? 1 : currentDailyCount + 1
  const newIds = [...existingIds, articleId]

  // 条件更新——只有当 ids 未变化时才更新（原子性保证）
  const { data: updated, error: upsertError } = await supabaseAdmin
    .from("user_profiles")
    .update({
      notes_read_ids: newIds,
      daily_read_count: newDailyCount,
      last_read_date: today,
    })
    .eq("id", userId)
    .eq("notes_read_ids", existingIds) // ← 原子性关键
    .select("notes_read_ids, daily_read_count, last_read_date")
    .single()

  if (upsertError) {
    // 更新失败（并发冲突），返回当前状态而非错误数据
    console.error("[ReadingLimit] 更新失败:", upsertError)
    return {
      readCount: existingIds.length,
      readIds: existingIds,
      dailyReadCount: newDailyCount,
      lastReadDate: today,
    }
  }

  return {
    readCount: updated?.notes_read_ids?.length ?? existingIds.length,
    readIds: updated?.notes_read_ids ?? existingIds,
    dailyReadCount: updated?.daily_read_count ?? newDailyCount,
    lastReadDate: updated?.last_read_date ?? today,
  }
}

export type MembershipType = "none" | "monthly" | "yearly"

/**
 * 统一会员类型检测
 * @param vipTier — 数据库中的 vip_tier 值
 * @param membershipType — 可选，来自 memberships 表的 membership_type
 * @param endDate — 可选，来自 memberships 表的 end_date
 */
export function detectMembershipType(
  vipTier: string | null | undefined,
  membershipType?: string | null,
  endDate?: string
): MembershipType {
  // 优先使用 memberships 表的数据
  if (membershipType && endDate) {
    const raw = membershipType.toLowerCase().replace("_", "")
    const isExpired = new Date(endDate).getTime() < Date.now()
    if (isExpired) return "none"

    if (raw.includes("year")) return "yearly"
    if (raw.includes("monthly")) return "monthly"
  }

  // Fallback 到 vip_tier
  if (!vipTier) return "none"
  const tier = String(vipTier).toLowerCase().replace("_", "")

  if (tier === "none" || tier === "null") return "none"
  if (tier.includes("year") || tier.includes("annual")) return "yearly"
  if (tier.includes("monthly")) return "monthly"

  return "none"
}
