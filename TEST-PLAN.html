1. 问题回顾 - 为什么测试没有发现这些问题

最近修复的问题：

❌ 月卡每日限制未生效
问题：月卡用户阅读超过8篇时没有弹出限制提示
根因：客户端没有处理服务端返回的 DAILY_LIMIT_EXCEEDED 错误码
教训：服务端返回的错误码必须在客户端有对应的处理逻辑
❌ 月卡终身限制被错误取消
问题：月卡用户被错误地设置了无终身限制
根因：取消终身限制是对的，但实现方式不正确
教训：每个业务规则的变更都需要双向测试（服务端+客户端）
❌ 阅读限制硬编码
问题：阅读限制数字（3、8）在代码中硬编码
根因：没有设计配置系统，导致每次改需求都要改代码
教训：业务可配置项应该从一开始就有配置系统
✅ 月卡每日限制未生效 — 已修复
原问题：月卡用户阅读超过8篇时没有弹出限制提示
根因：客户端没有处理服务端返回的 DAILY_LIMIT_EXCEEDED 错误码
修复：服务端改为先 dry-run 推算是否超限，超限直接 403 不写 DB； 前端侧边栏守卫用 useDailyQuotaCheck hook，在 API 数据加载前放行（quotaLoaded 状态），加载后正确拦截并显示弹窗
涉及：app/api/articles/[id]/route.ts、hooks/use-daily-quota-check.ts、components/article-sidebar.tsx
✅ 月卡终身限制被错误取消 — 已修复
原问题：月卡用户被错误地设置了无终身限制
修复：月卡用户有每日限额（默认 8 篇，可配置），无终身限制。 年卡用户无限制。服务端和前端逻辑已统一。
✅ 阅读限制硬编码 — 已修复
原问题：阅读限制数字（3、8）为代码中硬编码
修复：新增 reading_settings 数据库表，支持后台配置； 新增 /api/reading-settings GET/PUT API； 前端新增 useReadingSettings hook（5 分钟缓存）
涉及：lib/reading-settings.ts、app/api/reading-settings/route.ts、hooks/use-reading-settings.ts、lib/constants.ts
✅ 服务端超限后仍写入 DB — 已修复
原问题：月卡用户被 403 拦截后 daily_read_count 仍 +1
修复：GET /api/articles/[id] 改为先 dry-run 推算 projectedDailyCount，超限时直接返回 403，不调用 recordUserRead
✅ 客户端重复写入阅读记录 — 已修复
原问题：服务端 recordUserRead 已写一次 DB，客户端 recordVisit effect 再写一次（双重写入）
修复：客户端 recordVisit 在调用 API 前先 fetch 一次 readIds，已包含该文章则跳过写入
✅ 验证码明文存储 — 已修复
原问题：pending_registrations 表明文存储密码和验证码
修复：新增 lib/security.ts，使用 SHA-256 哈希存储（salt: rfyr_salt_v1）

当前仍需注意的问题：

  • portfolio API 无认证 — 任何人可 CRUD 所有实盘记录（未修复）
  • Storage 接口无认证 — 任何人可上传文件（未修复）
  • HMAC_SECRET 未配置 — 伪造 token 机制存在安全风险（需配置环境变量）
  • 游客阅读限制 — 依赖 localStorage，无法防止真正滥用（设计限制）

10.1 Jest + Supertest API 测试

// tests/api/reading-limit.test.ts
import { describe, it, expect, beforeAll, afterAll } from '@jest/globals'
import request from 'supertest'

// 测试配置
const API_BASE = process.env.TEST_API_URL || 'http://localhost:3000'

describe('阅读限制 API 测试', () => {
  const testUserId = 'test-user-reading-limit'
  const testArticleId = 'test-article-1'
  
  // ====== 游客访问测试 ======
  
  describe('游客访问', () => {
    it('READ-001: 游客访问免费文章应返回内容', async () => {
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}`)
        .query({ access_level: 'free' })
      
      expect(res.status).toBe(200)
      expect(res.body.accessType).toBe('free')
      expect(res.body.isFreeArticle).toBe(true)
    })
    
    it('READ-002: 游客访问付费文章应返回401', async () => {
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}`)
        .query({ access_level: 'monthly' })
      
      expect(res.status).toBe(401)
      expect(res.body.code).toBe('REQUIRE_LOGIN')
    })
  })
  
  // ====== 月卡用户测试 ======
  
  describe('月卡用户阅读限制', () => {
    it('READ-005: 月卡超每日限制应返回DAILY_LIMIT_EXCEEDED', async () => {
      // 先阅读8篇
      for (let i = 0; i < 8; i++) {
        await request(API_BASE)
          .get(`/api/articles/${testArticleId}-${i}`)
          .set('Authorization', `Bearer ${testMonthUserToken}`)
      }
      
      // 第9篇应该被拒绝
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}-9`)
        .set('Authorization', `Bearer ${testMonthUserToken}`)
      
      expect(res.status).toBe(403)
      expect(res.body.code).toBe('DAILY_LIMIT_EXCEEDED')
      expect(res.body.limit).toBe(8)
    })
    
    it('READ-006: 月卡每日内访问应返回文章', async () => {
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}`)
        .set('Authorization', `Bearer ${testMonthUserToken}`)
      
      expect(res.status).toBe(200)
      expect(res.body.accessType).toBe('monthly')
      expect(res.body.limit).toBe(Infinity)
    })
    
    it('READ-004: 月卡访问年卡专属应返回403', async () => {
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}`)
        .set('Authorization', `Bearer ${testMonthUserToken}`)
        .query({ access_level: 'yearly' })
      
      expect(res.status).toBe(403)
      expect(res.body.code).toBe('YEARLY_REQUIRED')
    })
  })
  
  // ====== 非会员用户测试 ======
  
  describe('非会员用户阅读限制', () => {
    it('READ-007: 非会员超终身限制应返回LIMIT_EXCEEDED', async () => {
      // 先阅读3篇
      for (let i = 0; i < 3; i++) {
        await request(API_BASE)
          .get(`/api/articles/${testArticleId}-${i}`)
          .set('Authorization', `Bearer ${testNormalUserToken}`)
      }
      
      // 第4篇应该被拒绝
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}-3`)
        .set('Authorization', `Bearer ${testNormalUserToken}`)
      
      expect(res.status).toBe(403)
      expect(res.body.code).toBe('LIMIT_EXCEEDED')
    })
    
    it('READ-EDGE-003: 重复阅读不应扣减计数', async () => {
      const articleId = `repeat-test-${Date.now()}`
      
      // 阅读第一遍
      await request(API_BASE)
        .get(`/api/articles/${articleId}`)
        .set('Authorization', `Bearer ${testNormalUserToken}`)
      
      // 阅读第二遍（同一文章）
      const res = await request(API_BASE)
        .get(`/api/articles/${articleId}`)
        .set('Authorization', `Bearer ${testNormalUserToken}`)
      
      expect(res.status).toBe(200)
      // 第二次阅读不应增加计数
      expect(res.body.readCount).toBe(res.body.readCount)
    })
  })
  
  // ====== 年卡用户测试 ======
  
  describe('年卡用户无限制', () => {
    it('READ-003: 年卡访问年卡专属应允许', async () => {
      const res = await request(API_BASE)
        .get(`/api/articles/${testArticleId}`)
        .set('Authorization', `Bearer ${testYearUserToken}`)
        .query({ access_level: 'yearly' })
      
      expect(res.status).toBe(200)
      expect(res.body.accessType).toBe('yearly')
      expect(res.body.isUnlimited).toBe(true)
    })
  })
})

export async function createTestUser(type: 'normal' | 'monthly' | 'yearly') {
  const userId = `test-${type}-${Date.now()}`
  
  // 创建用户记录
  await supabase.from('users').insert({
    id: userId,
    email: `${userId}@test.com`,
    vip_tier: type === 'normal' ? 'none' : type
  })
  
  // 创建会员记录（如果是会员）
  if (type !== 'normal') {
    const endDate = new Date()
    if (type === 'monthly') {
      endDate.setDate(endDate.getDate() + 30)
    } else {
      endDate.setFullYear(endDate.getFullYear() + 1)
    }
    
    await supabase.from('memberships').insert({
      user_id: userId,
      membership_type: type === 'monthly' ? 'monthly_vip' : 'annual_vip',
      start_date: new Date().toISOString().split('T')[0],
      end_date: endDate.toISOString().split('T')[0],
      status: 'active'
    })
  }
  
  return userId
}
